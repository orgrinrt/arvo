### S1: `mock/crates/arvo-bits/src/bits.rs` (new)

`Bits<const N: u8>` opaque container plus `BitWidth` and `BitAccess` impls.

```rust
//! Opaque N-bit container. Non-arithmetic; compared by identity.
//!
//! Backs an N-bit value in a `u64` storage; only the low N bits are
//! addressable. `N` is compile-time asserted to lie in 1..=64.
//!
//! Primary consumer: `arvo-hash` `ContentHash` alias (`Bits<28>`).

use crate::traits::{BitAccess, BitWidth};

/// N-bit opaque bit-pattern. Construct via `Bits::new(raw)`; the
/// constructor masks to the low N bits. No arithmetic, ordering by
/// identity only.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord, Default)]
pub struct Bits<const N: u8>(u64);

impl<const N: u8> Bits<N> {
    const _BOUNDS: () = {
        assert!(N > 0, "Bits<0> is invalid; use Bool for 1-bit values");
        assert!(N <= 64, "Bits<N> requires N <= 64");
    };

    const MASK: u64 = if N == 64 { u64::MAX } else { (1u64 << N) - 1 };

    /// Construct a `Bits<N>` from raw bits, masking to the low N bits.
    pub const fn new(raw: u64) -> Self {
        let _ = Self::_BOUNDS;
        Self(raw & Self::MASK)
    }

    /// The underlying bit pattern, in the low N bits of the returned `u64`.
    pub const fn bits(self) -> u64 {
        self.0
    }
}

impl<const N: u8> BitWidth for Bits<N> {
    const WIDTH: arvo::USize = arvo::USize(N as usize);
}

impl<const N: u8> BitAccess for Bits<N> {
    fn bit(self, idx: arvo::USize) -> arvo::Bool {
        arvo::Bool(((self.0 >> idx.0) & 1) != 0)
    }

    fn with_bit_set(self, idx: arvo::USize) -> Self {
        Self::new(self.0 | (1u64 << idx.0))
    }

    fn with_bit_cleared(self, idx: arvo::USize) -> Self {
        Self::new(self.0 & !(1u64 << idx.0))
    }

    fn with_bit_toggled(self, idx: arvo::USize) -> Self {
        Self::new(self.0 ^ (1u64 << idx.0))
    }
}
```

### S7: `mock/crates/arvo-hash/src/algo.rs` (new)

```rust
//! Hash algorithm contracts. Streaming is canonical; oneshot is a
//! blanket-impl ext trait.
//!
//! Concrete algorithms implement `Hasher<N>`. The blanket impl of
//! `HasherExt<N>` over every `Hasher<N>` exposes a `hash(bytes)`
//! one-shot form with no extra work at the impl site.

use arvo_bits::Bits;

/// Streaming N-bit hasher. Feed bytes via `update`, finalise to a
/// `Bits<N>`.
///
/// Algorithms implement this trait. One-shot ergonomics come from
/// the `HasherExt<N>` ext trait via its blanket impl.
pub trait Hasher<const N: u8> {
    /// Feed a byte chunk into the hasher.
    fn update(&mut self, bytes: &[u8]);

    /// Consume the hasher and produce the final N-bit digest.
    fn finalize(self) -> Bits<N>;
}

/// One-shot convenience: update with all bytes, then finalize.
///
/// Blanket-implemented for every `Hasher<N>`. Consumers bind
/// `H: HasherExt<N>` (or `H: Hasher<N>` with the ext trait in
/// scope) to call `.hash(bytes)` on any hasher.
pub trait HasherExt<const N: u8>: Hasher<N> + Sized {
    /// Hash a complete byte slice in one pass.
    fn hash(mut self, bytes: &[u8]) -> Bits<N> {
        self.update(bytes);
        self.finalize()
    }
}

impl<H, const N: u8> HasherExt<N> for H where H: Hasher<N> + Sized {}
```

### S8: `mock/crates/arvo-bits/tests/bits.rs` (new)

//! Unit tests for `Bits<N>`: construction, masking, BitWidth,
//! BitAccess.

use arvo::{Bool, USize};
use arvo_bits::{BitAccess, BitWidth, Bits};

#[test]
fn new_masks_high_bits() {
    let b = Bits::<4>::new(0xFF);
    assert_eq!(b.bits(), 0x0F);
}

#[test]
fn new_full_width_preserves() {
    let b = Bits::<64>::new(0xFFFF_FFFF_FFFF_FFFF);
    assert_eq!(b.bits(), 0xFFFF_FFFF_FFFF_FFFF);
}

#[test]
fn width_matches_n() {
    assert_eq!(<Bits<28> as BitWidth>::WIDTH, USize(28));
    assert_eq!(<Bits<64> as BitWidth>::WIDTH, USize(64));
}

#[test]
fn bit_access_read() {
    let b = Bits::<8>::new(0b1010_0101);
    assert_eq!(b.bit(USize(0)), Bool(true));
    assert_eq!(b.bit(USize(1)), Bool(false));
    assert_eq!(b.bit(USize(7)), Bool(true));
}

#[test]
fn bit_access_set_clear_toggle() {
    let b = Bits::<8>::new(0b0000_0000);
    let set = b.with_bit_set(USize(3));
    assert_eq!(set.bits(), 0b0000_1000);
    let cleared = set.with_bit_cleared(USize(3));
    assert_eq!(cleared.bits(), 0b0000_0000);
    let toggled = set.with_bit_toggled(USize(3));
    assert_eq!(toggled.bits(), 0b0000_0000);
}

#[test]
fn equality_and_hash() {
    use core::hash::{BuildHasher, Hasher};
    let a = Bits::<16>::new(0x1234);
    let b = Bits::<16>::new(0x1234);
    assert_eq!(a, b);
    // Compile-check that Bits implements Hash; runtime behaviour is
    // covered by Rust's Hash contract.
    let _ = (a, b);
}

### S10: `mock/crates/arvo-hash/tests/algo.rs` (new)

```rust
//! Trait-bound sanity: a trivial Hasher<N> implementor compiles,
//! and HasherExt's blanket hash() method is callable.

use arvo_bits::Bits;
use arvo_hash::{Hasher, HasherExt};

/// Minimal test hasher: XORs each incoming byte into a running
/// 28-bit state. Not a real hash algorithm; only exists to prove
/// the trait surface is implementable.
#[derive(Default)]
struct XorHash28(u64);

impl Hasher<28> for XorHash28 {
    fn update(&mut self, bytes: &[u8]) {
        for &b in bytes {
            self.0 ^= b as u64;
        }
    }
    fn finalize(self) -> Bits<28> {
        Bits::<28>::new(self.0)
    }
}

#[test]
fn hasher_streaming_path() {
    let mut h = XorHash28::default();
    h.update(b"hello ");
    h.update(b"world");
    let got = h.finalize();
    let mut expected: u64 = 0;
    for &b in b"hello world" {
        expected ^= b as u64;
    }
    assert_eq!(got.bits(), Bits::<28>::new(expected).bits());
}

#[test]
fn hasher_ext_oneshot_blanket() {
    let got = XorHash28::default().hash(b"hello world");
    let mut expected: u64 = 0;
    for &b in b"hello world" {
        expected ^= b as u64;
    }
    assert_eq!(got.bits(), Bits::<28>::new(expected).bits());
}
```
